  \ifcase\number#1\or 1\or 2\or 3\or 4\or 5\or 6\or 7\or 8\or 9\or

  \hexnumber@{\numexpr#1/16\relax}\hexnumber@{\numexpr#1-16*(#1/16)\relax}%

  \two@hex@digits{\numexpr#1/256\relax}%
